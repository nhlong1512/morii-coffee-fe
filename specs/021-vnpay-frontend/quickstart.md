import { useAuthStore } from "@/stores/auth";
import { CreateVnpayPaymentUrlRequest, CreateVnpayPaymentUrlResponse } from "@/types/api";
import { ReconcileVnpayPaymentRequest, ReconcileVnpayPaymentResponse } from "@/types/api";

const API_BASE = process.env.NEXT_PUBLIC_API_BASE_URL;

function getAuthHeader(): Record<string, string> {
  const token = useAuthStore.getState().accessToken;
  return {
    "Content-Type": "application/json",
    "Authorization": `Bearer ${token}`
  };
}

export async function createVnpayPaymentUrl(
  request: CreateVnpayPaymentUrlRequest
): Promise<CreateVnpayPaymentUrlResponse> {
  const response = await fetch(`${API_BASE}/v1/payments/vnpay/payment-url`, {
    method: "POST",
    headers: getAuthHeader(),
    body: JSON.stringify(request)
  });
  
  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.error?.message || "Failed to create payment URL");
  }
  
  const json = await response.json();
  return json.data;
}

export async function reconcileVnpayPayment(
  request: ReconcileVnpayPaymentRequest
): Promise<ReconcileVnpayPaymentResponse> {
  const response = await fetch(`${API_BASE}/v1/payments/vnpay/reconcile`, {
    method: "POST",
    headers: getAuthHeader(),
    body: JSON.stringify(request)
  });
  
  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.error?.message || "Failed to reconcile payment");
  }
  
  const json = await response.json();
  return json.data;
}

async function handleCheckoutSubmit(data: CheckoutFormData) {
  try {
    const { paymentMethod } = useCheckoutStore.getState();
    
    if (paymentMethod === "VNPAY") {
      // Create VNPAY payment URL

        deliveryProvinceName: data.province,
        deliveryDistrictName: data.district,
        deliveryWardName: data.ward,
        deliveryAddressDetail: data.address,
        deliveryPhoneNumber: data.phone,
        shippingProviderId: data.shippingProviderId,
        expectedDeliveryDate: data.deliveryDate,
        serviceId: data.shippingServiceId,
      });
      
      // Store pending checkout in sessionStorage
      setPendingCheckout({
        provider: "Vnpay",
        checkoutDraftId: response.checkoutDraftId,
        providerSessionId: response.txnRef,
        expiresAtUtc: response.expiresAtUtc,
      });
      
      // Redirect to VNPAY
      window.location.assign(response.paymentUrl);
    } else if (paymentMethod === "STRIPE") {
      // Existing Stripe checkout logic
      const stripe = await getStripe();
      await stripe.redirectToCheckout({ sessionId });
    } else {
      // COD: proceed to checkout confirmation
      createOrder({ paymentMethod: "COD" });
    }
  } catch (error) {
    setError(t("error.payment_url_failed"));
  }
}

## Phase 4: Return Page & Polling (2 days)

### Step 4.1: Create Return Page Component

**File**: `src/components/checkout/vnpay-return-state.tsx`

```typescript
"use client";

import { useEffect, useState } from "react";
import { useTranslations } from "next-intl";
import { useRouter } from "next/navigation";
import { reconcileVnpayPayment } from "@/services/payment-service";
import { getPendingCheckout, clearPendingCheckout } from "@/lib/payment";
import { useCheckoutStore } from "@/stores/checkout";

interface VnpayReturnStateProps {
  txnRef?: string;
  result?: string;
}

export const VnpayReturnState: React.FC<VnpayReturnStateProps> = ({ txnRef, result }) => {
  const t = useTranslations();
  const router = useRouter();
  const [status, setStatus] = useState<"loading" | "success" | "failed" | "expired" | "invalid">("loading");
  const [orderId, setOrderId] = useState<string | null>(null);
  const [orderNumber, setOrderNumber] = useState<string | null>(null);
  const [elapsedSeconds, setElapsedSeconds] = useState(0);
  
  useEffect(() => {
    const startTime = Date.now();
    const MAX_POLL_SECONDS = 300;
    
    const poll = async () => {
      try {
        const pending = getPendingCheckout();
        if (!pending) {
          setStatus("invalid");
          return;
        }
        
        const elapsed = (Date.now() - startTime) / 1000;
        setElapsedSeconds(Math.floor(elapsed));
        
        // Stop conditions
        if (elapsed > MAX_POLL_SECONDS) {
          setStatus("expired");
          return;
        }
        
        if (new Date(pending.expiresAtUtc) < new Date()) {
          setStatus("expired");
          return;
        }
        
        // Query backend
        const response = await reconcileVnpayPayment({
          checkoutDraftId: pending.checkoutDraftId,
          txnRef: pending.providerSessionId,
        });
        
        if (response.paymentStatus === "Paid") {
          setStatus("success");
          setOrderId(response.orderId || null);
          setOrderNumber(response.orderNumber || null);
          
          // Clear cart and checkout state
          useCheckoutStore.setState({ cart: [] });
          clearPendingCheckout();
          
          // Navigate after brief delay
          setTimeout(() => {
            router.push(`/orders/${response.orderId}`);
          }, 1000);
          return;
        }
        
        if (response.paymentStatus === "Failed" || response.paymentStatus === "NotRequired") {
          setStatus("failed");
          clearPendingCheckout();
          return;
        }
        
        // Still pending, schedule next poll
        const delay = 2000 + Math.random() * 1000; // 2-3 seconds
        setTimeout(poll, delay);
      } catch (error) {
        console.error("Polling error:", error);
        // Retry on network error
        if (error instanceof Error && error.message.includes("network")) {
          setTimeout(poll, 2000);
        } else {
          setStatus("failed");
        }
      }
    };
    
    poll();
  }, []);
  
  if (status === "loading") {
    return <div>{t("checkout.vnpay_pending_verification")}</div>;
  }
  
  if (status === "success") {
    return (
      <div className="text-center py-8">
        <h2 className="text-xl font-semibold text-green-600">
          {t("checkout.vnpay_success")}
        </h2>
        <p>{t("checkout.order_created", { number: orderNumber })}</p>
      </div>
    );
  }
  
  if (status === "failed") {
    return (
      <div className="text-center py-8">
        <h2 className="text-xl font-semibold text-red-600">
          {t("checkout.vnpay_failed")}
        </h2>
        <button onClick={() => router.push("/checkout")} className="mt-4 btn btn-primary">
          {t("checkout.retry")}
        </button>
      </div>
    );
  }
  
  if (status === "expired") {
    return (
      <div className="text-center py-8">
        <h2 className="text-xl font-semibold text-yellow-600">
          {t("checkout.vnpay_expired")}
        </h2>
        <p>{t("checkout.payment_expired_message")}</p>
        <button onClick={() => router.push("/checkout")} className="mt-4 btn btn-primary">
          {t("checkout.start_new_order")}
        </button>
      </div>
    );
  }
  
  return (
    <div className="text-center py-8">
      <h2 className="text-xl font-semibold text-red-600">
        {t("checkout.vnpay_invalid")}
      </h2>
    </div>
  );
};
```
